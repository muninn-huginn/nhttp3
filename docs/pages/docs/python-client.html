Python Client

Async HTTP/3 client with multiplexed streams — no head-of-line blocking on parallel requests.

Basic GET

import nhttp3, asyncio

async def main():
    ep = await nhttp3.Endpoint.bind("0.0.0.0", 0)
    conn = await ep.connect("localhost", 4433)
    send, recv = await conn.open_bidi_stream()

    await send.write(b"GET / HTTP/3\r\n\r\n")
    await send.finish()

    data = await recv.read(65536)
    print(data)

asyncio.run(main())

Parallel Requests

# 10 concurrent requests — each on its own QUIC stream
async def fetch(conn, path):
    send, recv = await conn.open_bidi_stream()
    await send.write(f"GET {path}\r\n\r\n".encode())
    await send.finish()
    return await recv.read(65536)

results = await asyncio.gather(
    *[fetch(conn, f"/item/{i}") for i in range(10)]
)
